		//ADMIN_REGISTER//

CREATE TABLE  "ADMIN_REGISTER" 
   (	"NAME" VARCHAR2(40) NOT NULL ENABLE, 
	"EMAIL" VARCHAR2(40) NOT NULL ENABLE, 
	"PASSWORD" VARCHAR2(20) NOT NULL ENABLE, 
	 CONSTRAINT "ADMIN_REGISTER_PK" PRIMARY KEY ("EMAIL") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_ADMIN_REGISTER" 
  before insert on "ADMIN_REGISTER"               
  for each row  
begin   
  if :NEW."EMAIL" is null then 
    select "ADMIN_REGISTER_SEQ".nextval into :NEW."EMAIL" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_ADMIN_REGISTER" ENABLE;
